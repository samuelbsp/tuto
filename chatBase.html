Ceci est une classe de base qui sera utilisé par ChatLine et ChatUser.

Ainsi que un constructeur qui sera utilisé par toutes les classes.

    		

/* Ceci est la classe de base, utilisé par ChatLine et ChatUser */

class ChatBase{

    // Ce constructeur est utilisé par toutes les classes:

    public function __construct(array $options){

        foreach($options as $k=>$v){
            if(isset($this->$k)){
                $this->$k = $v;
            }
        }
    }
}
